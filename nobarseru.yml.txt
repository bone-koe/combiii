name: Livenobar Updater (Manual Only)

on:
  workflow_dispatch: # Gerbang manual via Token

permissions:
  contents: write

jobs:
  scrape-livenobar:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repo
        uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Install dependencies
        run: |
          # Playwright dihapus, hanya butuh library requests dasar!
          pip install requests tzdata

      - name: Run Fast API Scraper
        run: |
          echo "🚀 Menjalankan misil API Livenobar..."
          python livenobar.py

      - name: Commit & Push safely
        run: |
          git config user.email "github-actions[bot]@users.noreply.github.com"
          git config user.name "github-actions[bot]"

          if [ -f Livenobar_BoneTV.m3u8 ]; then
            git add Livenobar_BoneTV.m3u8
          else
            echo "💀 File M3U8 tidak ditemukan (0 Pertandingan LIVE). Skip commit."
            exit 0
          fi

          if git diff --cached --quiet; then
            echo "Tidak ada perubahan jadwal baru, skip commit."
            exit 0
          fi

          git commit -m "Update Livenobar playlist via API [skip ci]"

          if ! git push origin main; then
            echo "Push tertahan — melakukan rebase otomatis..."
            git pull --rebase origin main
            git push origin main
          fi
